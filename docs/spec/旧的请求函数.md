/**
 * 将对象转换为 query string
 * @param {Object} obj - 参数对象
 * @returns {string}
 */
function serializeParams(obj) {
  if (!obj || Object.keys(obj).length === 0) return "";
  const parts = [];
  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      const value = toValue(obj[key]);
      if (value !== null && value !== undefined) {
        // 处理数组参数
        if (Array.isArray(value)) {
          if (value.length > 0) {
            // 为数组中的每个元素创建重复的键值对
            value.forEach((item) => {
              parts.push(
                encodeURIComponent(key) + "=" + encodeURIComponent(item),
              );
            });
          }
        } else {
          parts.push(encodeURIComponent(key) + "=" + encodeURIComponent(value));
        }
      }
    }
  }
  return parts.length ? "?" + parts.join("&") : "";
}

/**
 * 生成请求ID
 * 格式: front_时间戳_随机字符串
 * @returns {string} 请求ID
 */
function generateRequestId() {
  const timestamp = Date.now();
  const randomStr = Math.random().toString(36).substring(2, 11);
  return `front_${timestamp}_${randomStr}`;
}

// 存储当前请求ID（用于关联多个请求）
let currentRequestId = null;

// Token 刷新相关变量
let isRefreshing = false;
let refreshSubscribers = [];

/**
 * 订阅 token 刷新完成
 * @param {Function} callback - 刷新完成后的回调函数
 */
function subscribeTokenRefresh(callback) {
  refreshSubscribers.push(callback);
}

/**
 * 通知所有订阅者 token 刷新完成
 * @param {string} token - 新的 token
 */
function onRefreshed(token) {
  refreshSubscribers.forEach((callback) => callback(token));
  refreshSubscribers = [];
}

/**
 * 刷新 token
 * @returns {Promise<string>} 新的 token
 */
async function refreshToken() {
  const userStore = useUserStore();

  // 如果已经在刷新，等待刷新完成
  if (isRefreshing) {
    return new Promise((resolve) => {
      subscribeTokenRefresh(resolve);
    });
  }

  isRefreshing = true;
  const refreshRequestId = generateRequestId();
  console.log(`🔄 [${refreshRequestId}] 开始刷新 token`);

  try {
    const oldToken = userStore.token;

    if (!oldToken) {
      throw new Error("没有可刷旧的Token");
    }

    // 使用刷新token接口
    const finalUrl = systemInfo.BaseUrl + "/auth/refresh";

    const res = await new Promise((resolve, reject) => {
      uni.request({
        url: finalUrl,
        method: "POST",
        data: { token: oldToken },
        header: {
          "Content-Type": "application/json",
          "X-Request-Id": refreshRequestId, // 添加请求ID
        },
        timeout: 10000,
        dataType: "json",
        success: (res) => {
          // 从响应头获取请求ID
          const responseRequestId =
            res.header?.["X-Request-Id"] ||
            res.header?.["x-request-id"] ||
            refreshRequestId;

          const { statusCode, data: responseData } = res;
          if (statusCode >= 200 && statusCode < 300) {
            if (responseData?.code === 0) {
              console.log(`✅ [${responseRequestId}] Token 刷新成功`);
              resolve(responseData.data);
            } else {
              const msg = responseData?.msg || "刷新Token失败";
              console.error(`❌ [${responseRequestId}] Token 刷新失败:`, msg);
              reject(new Error(msg));
            }
          } else {
            const msg = responseData?.message || `服务器错误 (${statusCode})`;
            console.error(`❌ [${responseRequestId}] Token 刷新HTTP错误:`, {
              statusCode,
              message: msg,
            });
            reject(new Error(msg));
          }
        },
        fail: (err) => {
          console.error(
            `❌ [${refreshRequestId}] Token 刷新网络错误:`,
            err.errMsg,
          );
          let msg = "网络请求失败";
          if (err.errMsg?.includes("timeout")) {
            msg = "请求超时，请稍后重试";
          } else if (err.errMsg?.includes("request:fail")) {
            msg = "网络连接异常";
          }
          reject(new Error(msg));
        },
      });
    });

    const newToken = res.token;

    // 更新 store 中的 token
    userStore.setToken(newToken);

    // 通知所有等待的请求
    onRefreshed(newToken);
    return newToken;
  } catch (error) {
    console.error(`❌ [${refreshRequestId}] Token 刷新失败:`, error);

    // 刷新失败，清空 token，让用户重新登录
    userStore.clearToken();

    // 跳转到首页重新登录
    uni.showToast({
      title: "登录已过期，请重新登录",
      icon: "none",
      duration: 2000,
    });

    // 延迟跳转，避免 toast 被立即覆盖
    setTimeout(() => {
      uni.reLaunch({
        url: "/pages/index/index",
      });
      // 再次尝试登录
      initLogin();
    }, 1500);

    throw error;
  } finally {
    isRefreshing = false;
  }
}

/**
 * 内部请求函数（不包含 token 刷新逻辑）
 */
function requestInternal(url, method = "GET", data = {}, customHeader = {}) {
  const userStore = useUserStore();

  // 生成请求ID
  const requestId = generateRequestId();
  currentRequestId = requestId;

  // 构建基础 header
  const baseHeader = {
    "Content-Type": "application/json",
    "X-Request-Id": requestId, // 添加请求ID到请求头
    ...customHeader,
  };

  // 登录状态下自动添加 Authorization（除非跳过认证）
  if (userStore.isLogin && userStore.token) {
    baseHeader["Authorization"] = `Bearer ${userStore.token}`;
  }

  // 处理 GET/DELETE 的查询参数
  let finalUrl = systemInfo.BaseUrl + url;
  let requestData = data;

  const upperMethod = method.toUpperCase();
  if (["GET", "DELETE"].includes(upperMethod)) {
    finalUrl += serializeParams(data);
    requestData = undefined; // GET/DELETE 不应有 body
  }

  // 记录请求开始
  console.log(`🚀 [${requestId}] ${method} ${url}`, data || "");

  return new Promise((resolve, reject) => {
    uni.request({
      url: finalUrl,
      method: upperMethod,
      data: requestData,
      header: baseHeader,
      timeout: 10000,
      dataType: "json",
      success: (res) => {
        // 从响应头获取请求ID（后端返回的）
        const responseRequestId =
          res.header?.["X-Request-Id"] ||
          res.header?.["x-request-id"] ||
          requestId;

        const { statusCode, data: responseData } = res;

        if (statusCode >= 200 && statusCode < 300) {
          if (responseData?.code === 0) {
            console.log(
              `✅ [${responseRequestId}] ${method} ${url} 成功`,
              responseData.data || "",
            );
            resolve(responseData.data);
          } else {
            const msg = responseData?.msg || "请求失败";
            console.error(
              `❌ [${responseRequestId}] ${method} ${url} 业务错误:`,
              {
                code: responseData?.code,
                message: msg,
                requestId: responseRequestId,
              },
            );
            // 不显示 toast，由外层处理
            reject({
              code: responseData?.code,
              message: msg,
              data: responseData,
              requestId: responseRequestId,
            });
          }
        } else {
          // 对于非2xx状态码，尝试解析后端返回的错误信息
          const msg = responseData?.message || `服务器错误 (${statusCode})`;
          console.error(
            `❌ [${responseRequestId}] ${method} ${url} HTTP错误:`,
            {
              statusCode,
              message: msg,
              requestId: responseRequestId,
            },
          );
          reject({
            code: responseData?.code || 9999,
            message: msg,
            data: responseData,
            requestId: responseRequestId,
          });
        }
      },
      fail: (err) => {
        console.error(`❌ [${requestId}] ${method} ${url} 网络错误:`, {
          error: err.errMsg,
          requestId,
        });
        let msg = "网络请求失败";
        if (err.errMsg?.includes("timeout")) {
          msg = "请求超时，请稍后重试";
        } else if (err.errMsg?.includes("request:fail")) {
          msg = "网络连接异常";
        }
        reject(new Error(msg));
      },
    });
  });
}

/**
 * 通用请求封装（包含 token 刷新逻辑）
 * @param {string} url - 接口路径（不包含 baseUrl）
 * @param {string} method - 请求方法，默认 'GET'
 * @param {Object} data - 请求体数据（POST/PUT）或查询参数（GET/DELETE）
 * @param {Object} customHeader - 自定义请求头
 * @returns {Promise}
 */
export async function request(
  url,
  method = "GET",
  data = {},
  customHeader = {},
) {
  return requestInternal(url, method, data, customHeader).catch(
    async (error) => {
      // 检查是否是 token 过期错误
      const isTokenExpired =
        error.code === 1002 &&
        (error.message?.includes("Token已过期") ||
          error.message?.includes("认证失败"));

      if (isTokenExpired) {
        console.log("Token 已过期，尝试刷新");
        try {
          // 刷新 token
          await refreshToken();

          // 使用新 token 重试请求
          console.log("使用新 token 重试请求");
          return requestInternal(url, method, data, customHeader);
        } catch (refreshError) {
          // 刷新失败，直接抛出错误
          throw refreshError;
        }
      }

      // 其他错误，显示 toast 并抛出
      if (error.message && !error.message.includes("网络请求失败")) {
        uni.showToast({
          title: error.message,
          icon: "none",
          duration: 2000,
        });
      }
      throw error;
    },
  );
}

export const post = (url, data = {}) => request(url, "POST", data);
export const put = (url, data = {}) => request(url, "PUT", data);

export const login = () => {
  return new Promise((resolve, reject) => {
    wx.login({
      success: (e) => {
        resolve(e.code);
      },
      fail: (e) => {
        reject(e);
      },
      timeout: 3000,
    });
  });
};

export const initLogin = async () => {
  console.log("初始化登录");
  const userStore = useUserStore();
  const sceneStore = useSceneStore();
  if (sceneStore.scene === wxMiniProgramScene["moments-single-page"]) {
    console.log("朋友圈单页模式不请求也不获取");
    return;
  }
  // 如果已经有 token，直接标记为登录就绪
  if (userStore.isLogin) {
    console.log("✅ 已登录，跳过获取 openid==》", userStore.token);
    // console.log("jwtDecode",jwtDecode);
    const res = decodeJwt(userStore.token);
    console.log("从jwt中获取的用户的userId为", res);
    userStore.setUserId(res?.userId || "");
    return;
  }
  // 标记登录开始
  userStore.startLogin();

  console.log("开始获取登录凭证");
  const code = await login().catch((e) => {
    uni.showToast({
      title: "登录失败：" + e.errMsg,
      icon: "none",
    });
    userStore.loginFailed();
    return null;
  });

  if (!code) {
    console.log("获取登录凭证失败");
    return;
  }
  try {
    const res = await request("/auth/login", "POST", { code });
    userStore.setToken(res.token);
    userStore.setUserId(res.userId);
    console.log("✅ 登录成功，openid 已绑定");
  } catch (error) {
    userStore.clearToken();
    userStore.loginFailed();
    uni.showToast({
      title: "登录失败：" + error.errMsg,
      icon: "none",
    });
  }
};
